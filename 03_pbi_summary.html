🎯 วัตถุประสงค์

def process_pbi(self, pbi_id):
    """ประมวลผล PBI และดึงข้อมูล test cases, tasks, bugs"""

    # ดึงข้อมูล PBI พร้อม relations
    url = f'{base_url}/{project}/_apis/wit/workItems/{pbi_id}?$expand=relations'
    pbi_data = requests.get(url, auth=self.auth).json()
    relations = pbi_data.get('relations', [])

    # ดึง test cases ที่ link กับ PBI
    test_cases = self.get_linked_test_cases(relations)
    # ตัวอย่าง: [{'id': 111, 'outcome': 'Passed', 'priority': 'High'}, ...]

    # ดึง tasks ที่ link กับ PBI
    tasks = self.get_linked_task_details(relations)

    # ดึง bugs ที่ link กับ PBI
    bugs = self.get_linked_bugs(relations, pbi_id)

    # คำนวณ metrics
    metrics = MetricsCalculator.calculate_pbi_metrics(tasks, test_cases, bugs, qa_users)
    # metrics = {'total_tests': 10, 'passed': 7, 'failed': 2, 'invalid_case': 0, 'not_start': 1, 'total_bugs': 2}

    # เก็บข้อมูลลงใน rtm_data
    self.rtm_data.append({
        'pbi_id': f'PBI-{pbi_id}',
        **metrics,
        'tasks': tasks,
        'test_cases': test_cases,
        'bugs': bugs,
        'tags': pbi_data['fields'].get('System.Tags', '')
    })

def get_test_case_result_with_sprint_filter(self, test_id):
    """ดึงผล test case และกรองตาม sprint"""

    test_data = requests.get(f'{base_url}/{project}/_apis/wit/workItems/{test_id}').json()
    fields = test_data.get('fields', {})
    work_item_type = fields.get('System.WorkItemType', '')

    # รับเฉพาะ AXONS Test Scenario v2
    if work_item_type not in ['Axons Test Scenario', 'AXONS Test Scenario v2']:
        return None

    # ข้าม test ที่ Obsoleted
    test_result = fields.get('Custom.Testresult', '')
    if test_result == 'Obsoleted':
        return None

    # กรองตาม sprint (iteration path ต้องตรงกับ sprint ปัจจุบัน)
    if self.target_path:
        iteration_path = fields.get('System.IterationPath', '')
        if iteration_path != self.target_path:
            return None  # ข้าม test ที่ไม่ได้อยู่ใน sprint นี้

    # แปลง Custom.Testresult เป็น outcome
    if test_result.lower() in ['passed', 'pass']:
        outcome = 'Passed'
    elif test_result.lower() in ['failed', 'fail']:
        outcome = 'Failed'
    elif test_result.lower() in ['invalid case', 'invalid']:
        outcome = 'Invalid case'
    else:
        outcome = 'Not start'

    return {
        'id': test_id,
        'outcome': outcome,
        'priority': fields.get('Custom.Prioritylevel', 'Medium'),
        'automation_status': fields.get('Custom.Automationteststatus', ''),
        'test_type': fields.get('Custom.Test_type', ''),
        'work_item_type': work_item_type
    }

def calculate_pbi_metrics(tasks, test_cases, bugs, assign_users=None):
    """คำนวณ metrics ของ PBI หนึ่งตัว"""

    counters = {'passed': 0, 'failed': 0, 'invalid_case': 0, 'not_start': 0}

    # นับผลของแต่ละ test case
    for test_case in test_cases:
        outcome = test_case.get('outcome', 'Not start')
        work_item_type = test_case.get('work_item_type', '')

        # นับเฉพาะ AXONS Test Scenario types
        if work_item_type in ['Axons Test Scenario', 'AXONS Test Scenario v2']:
            if outcome.lower() in ['passed', 'pass']:
                counters['passed'] += 1
            elif outcome.lower() in ['failed', 'fail']:
                counters['failed'] += 1
            elif outcome.lower() in ['invalid case', 'invalid']:
                counters['invalid_case'] += 1
            else:
                counters['not_start'] += 1

    # กรอง tasks เฉพาะของ QA (ถ้ามี assign_users)
    if assign_users:
        filtered_tasks = [t for t in tasks if t.get('assigned_to') in assign_users]
    else:
        filtered_tasks = tasks

    return {
        'total_tasks': len(filtered_tasks),
        'total_tests': len(test_cases),
        'total_bugs': len(bugs),
        **counters
    }
