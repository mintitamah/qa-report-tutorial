🐞 Bug Traceability

หัวข้อนี้แสดง traceability ระหว่าง PBI กับ Bugs — Level 3 ของ RTM พร้อม Unlinked Bugs

🎯 วัตถุประสงค์

แสดง Traceability Matrix ระหว่าง Requirements (PBI) และ Defects ที่พบในระหว่างการทดสอบ ครอบคลุมทั้ง Linked Bugs ที่เชื่อมโยงกับ PBI และ Unlinked Bugs ที่พบใน Sprint แต่ยังไม่ได้ผูกกับ Requirement ใด เพื่อให้ทีมสามารถติดตามและจัดการ Defects ได้อย่างครบถ้วน

📍 ที่มาของข้อมูล

Bug มี 2 ประเภท:
1. Linked Bugs = bug ที่ link กับ PBI ผ่าน System.LinkTypes.Hierarchy-Forward relation
2. Unlinked Bugs = bug ที่อยู่ใน Sprint แต่ไม่ได้ link กับ PBI ใดๆ (ค้นหาแยกต่างหาก)
ข้อมูลดึงจาก field ไหน
Bug IDBUG-{work_item_id}
Bug DescriptionSystem.Title
SeverityMicrosoft.VSTS.Common.Severity
Bug StatusSystem.State
Fixed StatusCustom.FixedStatus (1-4=Open, 5=Closed)

def get_linked_bugs(self, relations, pbi_id):
    """ดึง AXONS Bug ที่ link กับ PBI"""
    bugs = []
    for relation in relations:
        if 'workItems' in relation.get('url', ''):
            bug_id = relation['url'].split('/')[-1]
            bug_details = self.get_bug_details_with_sprint_filter(bug_id)
            if bug_details:
                bug_details['pbi_id'] = f'PBI-{pbi_id}'
                bugs.append(bug_details)
                self.bug_data.append(bug_details)  # เก็บไว้ใน global bug_data ด้วย
    return bugs

def get_bug_details_with_sprint_filter(self, work_item_id):
    """ดึงรายละเอียด bug และกรองตาม sprint"""
    work_data = requests.get(f'{base_url}/{project}/_apis/wit/workItems/{work_item_id}').json()
    fields = work_data.get('fields', {})

    # รับเฉพาะ AXONS Bug
    if fields.get('System.WorkItemType') != 'AXONS Bug':
        return None

    return {
        'id': f'BUG-{work_item_id}',
        'title': fields.get('System.Title', 'N/A'),
        'severity': fields.get('Microsoft.VSTS.Common.Severity', 'Medium'),
        'status': fields.get('System.State', 'Active'),
        'fixed_status': fields.get('Custom.FixedStatus', ''),
        'assigned_to': fields.get('System.AssignedTo', {}).get('displayName', 'Unassigned'),
    }

def search_unlinked_bugs_in_sprint(self):
    """ค้นหา bugs ใน Sprint ที่ไม่ได้ link กับ PBI"""

    # Query หา AXONS Bug ทั้งหมดใน Sprint
    query = {
        "query": f"""
        SELECT [System.Id] FROM WorkItems
        WHERE [System.WorkItemType] = 'AXONS Bug'
        AND [System.IterationPath] = '{self.target_path}'
        """
    }
    response = requests.post(f'{base_url}/{project}/_apis/wit/wiql', json=query)
    all_bug_ids = [item['id'] for item in response.json().get('workItems', [])]

    # หา bug ที่ยังไม่ได้ link (ไม่อยู่ใน bug_data)
    linked_bug_ids = {bug['id'].replace('BUG-', '') for bug in self.bug_data}
    unlinked_bug_ids = [str(id) for id in all_bug_ids if str(id) not in linked_bug_ids]

    # ดึงรายละเอียดของ unlinked bugs
    for bug_id in unlinked_bug_ids:
        bug_details = self._extract_bug_details(work_data)
        if bug_details:
            bug_details['pbi_id'] = 'N/A'  # ไม่มี PBI
            self.unlinked_bugs.append(bug_details)

def generate_requirements_bugs_traceability(rtm_data, bug_data, org, project_name, ...):
    """สร้างตาราง Bug Traceability"""

    for item in rtm_data:
        pbi_id = item['pbi_id']
        # หา bugs ที่ link กับ PBI นี้
        pbi_bugs = [bug for bug in bug_data if bug.get('pbi_id') == pbi_id]

        if pbi_bugs:
            # แสดง PBI ID ที่ row แรก, row ถัดไปเว้นว่าง
            first_bug = pbi_bugs[0]
            html += f'<tr><td>{pbi_id}</td><td>{first_bug["id"]}</td>...</tr>'

            for bug in pbi_bugs[1:]:
                html += f'<tr><td></td><td>{bug["id"]}</td>...</tr>'  # PBI cell ว่าง
        else:
            html += f'<tr><td>{pbi_id}</td><td>—</td><td>No bugs found yet</td>...</tr>'

    # แสดง Unlinked Bugs แยกส่วน
    if unlinked_bugs:
        html += '<tr><td colspan="5">Unlinked Bugs (Not Associated with Any PBI)</td></tr>'
        for bug in unlinked_bugs:
            html += f'<tr><td>—</td><td>{bug["id"]}</td>...</tr>'
