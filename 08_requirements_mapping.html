def generate_requirements_mapping(rtm_data, query_mode, org, project_name, get_pbi_title_fn, ...):
    """สร้างตาราง Requirements Mapping"""

    for item in rtm_data:
        pbi_id = item['pbi_id']
        test_cases = item.get('test_cases', [])
        tags = item.get('tags', '')

        # ข้าม PBI ที่มี tag PBI.Validator:FAIL
        tag_list = [t.strip() for t in tags.split(';')] if tags else []
        has_validator_fail = any('pbi.validator:fail' in tag.lower() for tag in tag_list)
        if has_validator_fail:
            continue  # ข้าม PBI นี้

        # ดึง PBI Title จาก cache
        pbi_title = get_pbi_title_fn(pbi_id)  # System.Title

        # สร้าง links สำหรับแต่ละ test scenario
        tc_links = []
        for tc in test_cases:
            tc_id = tc.get('id', '')
            tc_url = f"https://dev.azure.com/{org}/{project}/_workitems/edit/{tc_id}"
            tc_links.append(f'<a href="{tc_url}">TC-{tc_id}</a>')

        # กำหนด Coverage Status
        if len(test_cases) > 0:
            coverage_status = '✅ Covered'
            coverage_style = 'color: #059669; font-weight: 600;'
        else:
            coverage_status = '❌ Not covered'
            coverage_style = 'color: #ef4444; font-weight: 600;'

        # สร้าง table row
        html += f'''
        <tr>
            <td><a href="{pbi_url}">{pbi_id}</a></td>
            <td>{pbi_title}</td>
            <td>{len(test_cases)}</td>
            <td>{', '.join(tc_links) or '-'}</td>
            <td style="{coverage_style}">{coverage_status}</td>
        </tr>
        '''
