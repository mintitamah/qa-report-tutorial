ตัวเลข % แสดงสัดส่วนของ test scenarios ใน priority นั้น เทียบกับ test ทั้งหมดของ Sprint
เช่น High 98.8% หมายความว่า test ส่วนใหญ่ใน Sprint นี้เป็น High priority

🎨 นิยามสีของแต่ละ Priority

Priorityสีความหมายตัวอย่าง Feature
Critical #ef4444 (แดง) ต้องทดสอบก่อนเลย — ถ้า fail ระบบใช้งานไม่ได้ Login, Payment, Core API
High #f97316 (ส้ม) สำคัญมาก — ส่งผลต่อ business logic หลัก Report generation, Data sync
Medium #eab308 (เหลือง) ปานกลาง — ส่งผลต่อ feature รอง UI validation, Filter
Low #22c55e (เขียว) ต่ำ — cosmetic หรือ nice-to-have Tooltip, Color theme

def _calculate_priority_stats(rtm_data, assign_users=None):
    """คำนวณสถิติ test scenarios แยกตาม priority"""

    stats = {'Critical': 0, 'High': 0, 'Medium': 0, 'Low': 0}
    total_all_tests = 0

    for item in rtm_data:
        for test_case in item.get('test_cases', []):
            priority = test_case.get('priority', 'Medium')  # Custom.Prioritylevel
            if priority in stats:
                stats[priority] += 1
            total_all_tests += 1

    # ผลลัพธ์: {'Critical': (0%, 0, 511), 'High': (98.8%, 481, 487), ...}
    # format: (percentage, count, total)
    return {
        p: (count / total_all_tests * 100 if total_all_tests > 0 else 0, count, total_all_tests)
        for p, count in stats.items()
    }

def generate_priority_coverage(rtm_data, config=None, assign_users=None):
    """สร้าง HTML แสดง Test Scenario Priority Coverage"""

    priority_colors = {
        'Critical': '#ef4444',  # แดง
        'High':     '#f97316',  # ส้ม
        'Medium':   '#eab308',  # เหลือง
        'Low':      '#22c55e'   # เขียว
    }

    stats = _calculate_priority_stats(rtm_data, assign_users)

    items = []
    for priority in ['Critical', 'High', 'Medium', 'Low']:
        coverage, count, total = stats.get(priority, (0, 0, 0))
        color = priority_colors[priority]
        items.append(f'''
        

            
                {round(coverage, 1)}%
            
            
{priority}

            
{count}/{total} test scenarios

        

        ''')

    return f'
{" ".join(items)}
'
