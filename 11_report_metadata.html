def generate_metadata(data, config=None):
    """สร้าง Report Metadata section"""

    # ดึง sprint scope จาก data
    current_sprint = data.get('current_sprint', '')
    sprint_scope = current_sprint if data.get('sprint_name') else 'Current Active Sprint'
    # ตัวอย่าง: "AgriTech\Dragon Ball\Sprint 97"

    # ดึงวันที่จาก data
    report_date = data.get('report_date', '')
    # ตัวอย่าง: "2025-01-15 14:30:22"

    # ค่าคงที่จาก config
    metadata_content = config.get('metadata_content', {})
    # metadata_content = {
    #     'report_type': 'Requirements Traceability Matrix (RTM) with Bug Analysis',
    #     'data_source': 'Azure DevOps Work Items and Test Cases',
    #     'generated_by': 'Amazon Q AXONS Report Generator'
    # }

    return f'''
    <section class="section">
        <h2>📄 Report Metadata</h2>
        <div class="summary-grid">
            <div class="summary-card">
                <strong>Report Type</strong>
                {metadata_content['report_type']}
            </div>
            <div class="summary-card">
                <strong>Data Source</strong>
                {metadata_content['data_source']}
            </div>
            <div class="summary-card">
                <strong>Sprint Scope</strong>
                {sprint_scope}
            </div>
            <div class="summary-card">
                <strong>Last Updated</strong>
                {report_date}
            </div>
            <div class="summary-card">
                <strong>Generated By</strong>
                {metadata_content['generated_by']}
            </div>
        </div>
    </section>
    '''
