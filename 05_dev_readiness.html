📍 ที่มาของข้อมูล

หลักการแยก Dev Tasks จาก QA Tasks:
• QA Tasks = tasks ที่ assigned_to อยู่ใน assign_users list (config)
• Dev Tasks = tasks ที่ assigned_to ไม่อยู่ใน assign_users list
• Tasks ที่ status = "Removed" จะถูกข้ามทั้งหมด
• Open Bug = Custom.FixedStatus ∈ {1, 2, 3, 4} (5 = Closed)

🧠 Logic การตัดสิน Readiness

def calculate_dev_readiness(rtm_data, qa_users):
    """คำนวณ readiness ของแต่ละ PBI"""

    for pbi in rtm_data:
        # แยก dev tasks (ไม่ใช่ QA และไม่ใช่ Removed)
        dev_tasks = [
            t for t in pbi.get('tasks', [])
            if t.get('assigned_to') not in qa_users
            and t.get('status', '').lower() not in ['removed']
        ]

        tasks_done  = sum(1 for t in dev_tasks if t.get('status', '').lower() in ['done', 'completed'])
        tasks_total = len(dev_tasks)

        # open bugs = Custom.FixedStatus ∈ {1, 2, 3, 4}
        open_bugs = [b for b in pbi.get('bugs', []) if b.get('fixed_status', '').strip() in ['1', '2', '3', '4']]

        dev_ready  = (tasks_done == tasks_total) if tasks_total > 0 else True
        bugs_ready = len(open_bugs) == 0

        if dev_ready and bugs_ready:
            ready_pbis.append(pbi)        # ✅ PBI Ready for QA
        elif not dev_ready:
            blocked_by_dev.append(pbi)    # 🔴 PBI Blocked by Dev Tasks
        elif not bugs_ready:
            blocked_by_bugs.append(pbi)   # 🐛 PBI Blocked by Bugs
