def generate_recommendations_section(rtm_data, bug_status_count, total_bugs):
    """สร้าง Recommendations section"""

    # คำนวณ pending tests (Not Start)
    total_not_run = sum(item.get('not_run', 0) for item in rtm_data)

    # คำนวณ open bugs (ทุก status ยกเว้น Done)
    open_bugs_count = (
        bug_status_count.get('To Do', 0) +
        bug_status_count.get('In Progress', 0) +
        bug_status_count.get('Impediment', 0) +
        bug_status_count.get('Code Review', 0)
    )

    html_content = ''

    # เงื่อนไขที่ 1: มี test ที่ยังไม่เริ่ม
    if total_not_run > 0:
        html_content += f'''
        <div class="summary-card">
            🎯 <strong>Execute {total_not_run} pending test cases</strong> to improve coverage
        </div>
        '''

    # เงื่อนไขที่ 2: มี open bugs
    if open_bugs_count > 0:
        html_content += f'''
        <div class="summary-card">
            🐛 <strong>Resolve {open_bugs_count} open bugs</strong> before sprint completion
        </div>
        '''

    # เงื่อนไขที่ 3: มี Impediment bugs (ด่วนมาก)
    if bug_status_count.get('Impediment', 0) > 0:
        html_content += f'''
        <div class="summary-card">
            🚨 <strong>Address {bug_status_count["Impediment"]} impediment bugs</strong> immediately
        </div>
        '''

    # เงื่อนไขที่ 4: มี Code Review bugs
    if bug_status_count.get('Code Review', 0) > 0:
        html_content += f'''
        <div class="summary-card">
            👀 <strong>Review {bug_status_count["Code Review"]} bugs in code review</strong>
        </div>
        '''

    # เงื่อนไขที่ 5: ไม่มี bug เลย
    if total_bugs == 0:
        html_content += '''
        <div class="summary-card">
            ✅ <strong>Excellent quality!</strong> No bugs found
        </div>
        '''

    # เงื่อนไขที่ 6: ทุกอย่างเรียบร้อย
    if total_not_run == 0 and open_bugs_count == 0:
        html_content += '''
        <div class="summary-card">
            🎉 <strong>Sprint is on track!</strong> All metrics are within acceptable ranges
        </div>
        '''

    return html_content

🔑 Keyword สำคัญ

Keywordความหมายค่าที่เป็นไปได้
not_runจำนวน test ที่ยังไม่เริ่มinteger ≥ 0
bug_status_countdictionary นับ bug แยกตาม status{'To Do': 2, 'In Progress': 1, 'Impediment': 0, 'Code Review': 1, 'Done': 3}
Impedimentbug ที่ติดปัญหา ต้องแก้ด่วนSystem.State = Impediment
Code Reviewbug ที่รอ review codeSystem.State = Code Review
open_bugs_countbug ที่ยังไม่ DoneTo Do + In Progress + Impediment + Code Review
