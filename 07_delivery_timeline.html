📅 Dev Tasks Delivery Timeline

หัวข้อนี้แสดง timeline ว่า Dev ส่งมอบงานให้ QA วันไหนบ้างใน Sprint — ช่วยวิเคราะห์ว่า Dev ส่งงานทันเวลาหรือไม่

🎯 วัตถุประสงค์

แสดง Timeline การส่งมอบงานของทีม Development ตลอด Sprint (10 Work Days) โดยระบุวันที่ Dev Tasks แต่ละรายการถูก Complete เพื่อวิเคราะห์รูปแบบการส่งมอบงาน และประเมินผลกระทบต่อระยะเวลาที่ทีม QA มีสำหรับการทดสอบ

Day 1–7 = On-time — Dev ส่งมอบงานภายในกรอบเวลาที่กำหนด ทีม QA มีเวลาทดสอบเพียงพอ
Day 8–10 = Delay — Dev ส่งมอบงานล่าช้า ส่งผลให้ทีม QA มีเวลาทดสอบไม่เพียงพอ

🔍 Filter PBI ที่มี Test Scenario เท่านั้น

เหตุผล: PBI ที่ไม่มี test scenario = ไม่ใช่งาน developing จริงๆ (อาจเป็น support, documentation, etc.)
tasks ของ PBI เหล่านั้นไม่ควรนับใน Delay/Timeline

เงื่อนไข: if pbi.get('total_tests', 0) == 0: continue
ใช้ใน calculate_dev_tasks_by_day() และ calculate_dev_tasks_delay()

💻 โค้ดหลักที่ใช้

def calculate_dev_tasks_by_day(rtm_data, qa_users, sprint_start_date, total_sprint_days=10):
    """แมป dev tasks ไปยังแต่ละวันทำงานของ Sprint
    นับเฉพาะ tasks จาก PBI ที่มี test scenario (total_tests > 0)"""

    work_days = []
    current = sprint_start_date
    while len(work_days) < total_sprint_days:
        if current.weekday() < 5:  # 0=จันทร์, 4=ศุกร์
            work_days.append(current)
        current += timedelta(days=1)

    days_map = {i + 1: {'date': work_days[i], 'tasks': []} for i in range(len(work_days))}

    for pbi in rtm_data:
        if pbi.get('total_tests', 0) == 0:
            continue  # ข้าม PBI ที่ไม่มี test scenario

        dev_tasks = [
            t for t in pbi.get('tasks', [])
            if t.get('assigned_to') not in qa_users
            and t.get('status', '').lower() in ['done', 'completed']
        ]

        for task in dev_tasks:
            changed_date = datetime.fromisoformat(task.get('changed_date', ''))
            for day_num, day_info in days_map.items():
                if changed_date.date() == day_info['date'].date():
                    days_map[day_num]['tasks'].append(task)
                    break

    return days_map


def calculate_dev_tasks_delay(rtm_data, qa_users, sprint_start_date, delay_threshold_day=8):
    """นับ dev tasks ที่เสร็จหลัง Day 8 (DELAY)
    นับเฉพาะ tasks จาก PBI ที่มี test scenario (total_tests > 0)"""

    threshold_date = sprint_start_date
    days_added = 0
    while days_added < delay_threshold_day - 1:
        threshold_date += timedelta(days=1)
        if threshold_date.weekday() < 5:
            days_added += 1

    delay_count = 0
    for pbi in rtm_data:
        if pbi.get('total_tests', 0) == 0:
            continue  # ข้าม PBI ที่ไม่มี test scenario

        dev_tasks = [t for t in pbi.get('tasks', []) if t.get('assigned_to') not in qa_users
                     and t.get('status', '').lower() in ['done', 'completed']]
        for task in dev_tasks:
            changed_date = datetime.fromisoformat(task.get('changed_date', ''))
            if changed_date >= threshold_date:
                delay_count += 1

    return {'delay_count': delay_count}
